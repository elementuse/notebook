# R通道
u, sigma, v = np.linalg.svd(ori_img[:, :, 0])
r = np.dot(np.dot(u[:,:k], np.eye(k) * sigma[:k]), v[:k,:])
r[r<0]=0
r[r>255]=255
r=np.rint(r).astype("uint8")
