from PIL import Image
import matplotlib.pyplot as plt

# 读取图片
filename = 'cat.jpg';
ori_img = np.array(Image.open(filename).convert('L'))
In [3]:
# SVD分解
u, sigma, vt = np.linalg.svd(ori_img)
In [4]:
# 将奇异值向量的前k个奇异值转为对角矩阵
k = 10
sig = np.eye(k) * sigma[:k]

# 生成压缩图片
img1 = np.dot(np.dot(u[:,:k], sig), vt[:k,:])
